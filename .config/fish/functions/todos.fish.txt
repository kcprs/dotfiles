function unln() {
    for unln_path in "$@"; do
        local unln_name=$(basename "$unln_path")
        local unln_tmp="/tmp/$unln_name"

        # Using directly from /bin to avoid problems with `alias cp="cp -i"`
        /bin/cp -Lf "$unln_path" "$unln_tmp"
        /bin/mv -f "$unln_tmp" "$unln_path"
    done
}
